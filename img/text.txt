<script type="text/javascript">
	$(function () {
		$('#a1').change(function () {
			tmyFunction();
		});
		$('#a2').change(function () {
			tmyFunction();
		});
		$('#a3').change(function () {
			tmyFunction();
		});
	});
	function tmyFunction(){
		var a1 = $('#a1').maskMoney('unmasked')[0];
		var a2 = $('#a2').val();
		var a3 = $('#a3').val();
		a2 = (a2*1);
		a3 = (a3*1);

		//----------------------------------------------------------------------------------------------------------------
		//---anh sửa công thức lại là được
		//---a1 = tiền đề xuất
		//---a2 = tháng vay
		//---a3 = lãi xuất
		//---t1 = kết quả

		t1= a1 + a2 +a3;
		
		var string = numeral(t1).format('0,0.00');
		$('#t5').val(string +" Triệu VNĐ");
	}
</script>
